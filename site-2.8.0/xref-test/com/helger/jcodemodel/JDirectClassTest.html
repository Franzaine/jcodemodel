17    @Test
18    public void testBasic () throws Exception
19    {
20      final JCodeModel cm = new JCodeModel ();
21  
22      final JDirectClass rClassId = cm.directClass ("id.aa.R")._class ("id");
23      assertEquals ("id", rClassId.name ());
24      assertEquals ("id.aa", rClassId._package ().name ());
25      assertEquals ("id.aa.R.id", rClassId.fullName ());
26      final JDirectClass rClassMenu = cm.directClass ("id.aa.R")._class ("menu");
27      assertEquals ("menu", rClassMenu.name ());
28      assertEquals ("id.aa", rClassMenu._package ().name ());
29      assertEquals ("id.aa.R.menu", rClassMenu.fullName ());
30  
31      final JFieldRef myItem = rClassId.staticRef ("myItem");
32      final JFieldRef myMenu = rClassMenu.staticRef ("myMenu");
33  
34      final JPackage aPkg2 = cm._package ("id.aa");
35      final JDefinedClass aClassAct = aPkg2._class ("HelloAndroidActivity_");
36      final JMethod aMethodCreate = aClassAct.method (JMod.PUBLIC, cm.BOOLEAN, "onCreateOptionsMenu");
37      aMethodCreate.body ().add (JExpr.ref ("menuInflater").invoke ("inflate").arg (myMenu));
38      final JMethod aMethodSelected = aClassAct.method (JMod.PUBLIC, cm.BOOLEAN, "onOptionsItemSelected");
39      aMethodSelected.body ()._if (JExpr.ref ("itemId_").eq (myItem));
40  
41      CodeModelTestsHelper.parseCodeModel (cm);
42    }

44    @Test
45    public void testGenerics () throws Exception
46    {
47      final JCodeModel cm = new JCodeModel ();
48  
49      final AbstractJClass aNarrowedClass = (AbstractJClass) cm.parseType ("com.test.GenericFragmentArguments<S,P>");
50      assertTrue (aNarrowedClass instanceof JNarrowedClass);
51      assertTrue (aNarrowedClass.erasure () instanceof JDirectClass);
52      assertEquals ("com.test", aNarrowedClass._package ().name ());
53      assertEquals ("GenericFragmentArguments<S,P>", aNarrowedClass.name ());
54      assertEquals ("GenericFragmentArguments", aNarrowedClass.erasure ().name ());
55      assertEquals ("com.test.GenericFragmentArguments<S,P>", aNarrowedClass.fullName ());
56      assertEquals ("com.test.GenericFragmentArguments", aNarrowedClass.erasure ().fullName ());
57  
58      cm._class ("UsingClass").method (JMod.PUBLIC, cm.VOID, "test").body ().add (JExpr._new (aNarrowedClass));
59  
60      CodeModelTestsHelper.parseCodeModel (cm);
