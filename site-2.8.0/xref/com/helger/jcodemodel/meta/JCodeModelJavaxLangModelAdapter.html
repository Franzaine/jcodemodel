30  package com.helger.jcodemodel.meta;
31  
32  import javax.lang.model.element.TypeElement;
33  import javax.lang.model.util.Elements;
34  
35  import com.helger.jcodemodel.JCodeModel;
36  import com.helger.jcodemodel.JDefinedClass;
37  
38  public class JCodeModelJavaxLangModelAdapter
39  {
40    private final JCodeModel _codeModel;
41    private final Elements _elementUtils;
42  
43    /**
44     * Creates new instance of JCodeModelJavaxLangModelAdapter.
45     */
46    public JCodeModelJavaxLangModelAdapter (final JCodeModel codeModel, final Elements elementUtils)
47    {
48      this._codeModel = codeModel;
49      this._elementUtils = elementUtils;
50    }
51  
52    /**
53     * Returns jcodemodel class definition for given element.
54     *
55     * @param element
56     *        element to convert to class definition
57     * @throws ErrorTypeFound
58     *         if {@code element} argument contains references to so called
59     *         "error"-types.
60     */
61    public JDefinedClass getClass (final TypeElement element) throws ErrorTypeFound, CodeModelBuildingException
62    {
63      final DecidedErrorTypesModelsAdapter errorTypeDecision = new DecidedErrorTypesModelsAdapter (_codeModel,
64                                                                                                   _elementUtils,
65                                                                                                   false);
66      try
67      {
68        return errorTypeDecision.getClass (element);
69      }
70      catch (final RuntimeErrorTypeFound ex)
71      {
72        throw ex.getCause ();
73      }
74    }
75  
76    /**
77     * Returns jcodemodel class definition for given element.
78     * <p>
79     * This method result-class definition can include references to
80     * "error"-types. Error-types are used only if they are present in
81     * {@code element} argument
82     *
83     * @param element
84     *        element to convert to class definition
85     */
86    public JDefinedClass getClassWithErrorTypes (final TypeElement element) throws CodeModelBuildingException
87    {
88      final DecidedErrorTypesModelsAdapter errorTypeDecision = new DecidedErrorTypesModelsAdapter (_codeModel,
89                                                                                                   _elementUtils,
90                                                                                                   true);
91      return errorTypeDecision.getClass (element);
92    }
93  
94  }
