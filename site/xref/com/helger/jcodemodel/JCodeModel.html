64  /**
65   * Root of the code DOM.
66   * <p>
67   * Here's your typical CodeModel application.
68   * 
69   * <pre>
70   * JCodeModel cm = new JCodeModel();
71   * 
72   * // generate source code by populating the 'cm' tree.
73   * cm._class(...);
74   * ...
75   * 
76   * // write them out
77   * cm.build(new File("."));
78   * </pre>
79   * <p>
80   * Every CodeModel node is always owned by one {@link JCodeModel} object at any
81   * given time (which can be often accesesd by the <tt>owner()</tt> method.) As
82   * such, when you generate Java code, most of the operation works in a top-down
83   * fashion. For example, you create a class from {@link JCodeModel}, which gives
84   * you a {@link JDefinedClass}. Then you invoke a method on it to generate a new
85   * method, which gives you {@link JMethod}, and so on. There are a few
86   * exceptions to this, most notably building {@link IJExpression}s, but
87   * generally you work with CodeModel in a top-down fashion. Because of this
88   * design, most of the CodeModel classes aren't directly instanciable.
89   * <h2>Where to go from here?</h2>
90   * <p>
91   * Most of the time you'd want to populate new type definitions in a
92   * {@link JCodeModel}. See {@link #_class(String, EClassType)}.
93   */
94  public final class JCodeModel

108   static
109   {
110     final Map <Class <?>, Class <?>> m1 = new HashMap <Class <?>, Class <?>> ();
111     final Map <Class <?>, Class <?>> m2 = new HashMap <Class <?>, Class <?>> ();
112 
113     m1.put (Boolean.class, Boolean.TYPE);
114     m1.put (Byte.class, Byte.TYPE);
115     m1.put (Character.class, Character.TYPE);
116     m1.put (Double.class, Double.TYPE);
117     m1.put (Float.class, Float.TYPE);
118     m1.put (Integer.class, Integer.TYPE);
119     m1.put (Long.class, Long.TYPE);
120     m1.put (Short.class, Short.TYPE);
121     m1.put (Void.class, Void.TYPE);
122 
123     // Swap keys and values
124     for (final Map.Entry <Class <?>, Class <?>> e : m1.entrySet ())
125       m2.put (e.getValue (), e.getKey ());
126

128     primitiveToBox = Collections.unmodifiableMap (m2);
129   }

161   protected boolean getFileSystemCaseSensitivity ()
162   {
163     try
164     {
165       // let the system property override, in case the user really
166       // wants to override.
167       if (System.getProperty ("com.sun.codemodel.FileSystemCaseSensitive") != null)
168         return true;
169 
170       // Add special override to differentiate if Sun implementation is also in
171       // scope
172       if (System.getProperty ("com.helger.jcodemodel.FileSystemCaseSensitive") != null)
173         return true;
174     }
175     catch (final Exception e)
176     {}
177 
178     // on Unix, it's case sensitive.
179     return File.separatorChar == '/';
180   }

375   /**
376    * Generates Java source code.
377    */
378   public void build (@Nonnull final AbstractCodeWriter source, @Nonnull final AbstractCodeWriter resource) throws IOException
379   {
380     try
381     {
382       final JPackage [] pkgs = _packages.values ().toArray (new JPackage [_packages.size ()]);
383       // avoid concurrent modification exception
384       for (final JPackage pkg : pkgs)
385         pkg.build (source, resource);
386     }
387     finally
388     {
389       source.close ();
390       resource.close ();
391     }
392   }

409   /**
410    * Obtains a reference to an existing class from its Class object.
411    * <p>
412    * The parameter may not be primitive.
413    * 
414    * @see #_ref(Class) for the version that handles more cases.
415    */
416   @Nonnull
417   public AbstractJClass ref (@Nonnull final Class <?> clazz)
418   {
419     JReferencedClass jrc = _refClasses.get (clazz);
420     if (jrc == null)
421     {
422       if (clazz.isPrimitive ())
423         throw new IllegalArgumentException (clazz + " is a primitive");
424       if (clazz.isArray ())
425         return new JArrayClass (this, _ref (clazz.getComponentType ()));
426       jrc = new JReferencedClass (clazz);
427       _refClasses.put (clazz, jrc);
428     }
429     return jrc;
430   }

440   /**
441    * Obtains a reference to an existing class from its fully-qualified class
442    * name.
443    * <p>
444    * First, this method attempts to load the class of the given name. If that
445    * fails, we assume that the class is derived straight from {@link Object},
446    * and return a {@link AbstractJClass}.
447    */
448   @Nonnull
449   public AbstractJClass ref (@Nonnull final String fullyQualifiedClassName)
450   {
451     try
452     {
453       // try the context class loader first
454       return ref (SecureLoader.getContextClassLoader ().loadClass (fullyQualifiedClassName));
455     }
456     catch (final ClassNotFoundException e)
457     {
458       // fall through
459     }
460 
461     // then the default mechanism.
462     try
463     {
464       return ref (Class.forName (fullyQualifiedClassName));
465     }
466     catch (final ClassNotFoundException e1)
467     {
468       // fall through
469     }
470 
471     // assume it's not visible to us.
472     return new JDirectClass (this, fullyQualifiedClassName);
473   }

487   /**
488    * Obtains a type object from a type name.
489    * <p>
490    * This method handles primitive types, arrays, and existing {@link Class}es.
491    * 
492    * @return The internal representation of the specified name
493    */
494   @Nonnull
495   public AbstractJType parseType (@Nonnull final String name)
496   {
497     // array
498     if (name.endsWith ("[]"))
499     {
500       // Simply remove trailing "[]"
501       return parseType (name.substring (0, name.length () - 2)).array ();
502     }
503 
504     // try primitive type
505     try
506     {
507       return AbstractJType.parse (this, name);
508     }
509     catch (final IllegalArgumentException e)
510     {
511       // NOt a primitive type
512     }
513 
514     // existing class
515     return new TypeNameParser (name).parseTypeName ();
516   }

518   @NotThreadSafe
519   private final class TypeNameParser
520   {
521     private final String _s;
522     private int _idx;
523 
524     public TypeNameParser (@Nonnull final String s)
525     {
526       _s = s;
527     }
528 
529     /**
530      * Parses a type name token T (which can be potentially of the form
531      * Tr&ly;T1,T2,...>, or "? extends/super T".)
532      * 
533      * @return The parsed type name
534      */
535     @Nonnull
536     AbstractJClass parseTypeName ()
537     {
538       final int start = _idx;
539 
540       if (_s.charAt (_idx) == '?')
541       {
542         // wildcard
543         _idx++;
544         _skipWs ();
545         final String head = _s.substring (_idx);
546         if (head.startsWith ("extends"))
547         {
548           _idx += 7;
549           _skipWs ();
550           return parseTypeName ().wildcard ();
551         }
552         if (head.startsWith ("super"))
553           throw new UnsupportedOperationException ("? super T not implemented");
554         // not supported
555         throw new IllegalArgumentException ("only extends/super can follow ?, but found " + _s.substring (_idx));
556       }
557 
558       while (_idx < _s.length ())
559       {
560         final char ch = _s.charAt (_idx);
561         if (Character.isJavaIdentifierStart (ch) || Character.isJavaIdentifierPart (ch) || ch == '.')
562           _idx++;
563         else
564           break;
565       }
566 
567       final AbstractJClass clazz = ref (_s.substring (start, _idx));
568 
569       return _parseSuffix (clazz);
570     }
571 
572     /**
573      * Parses additional left-associative suffixes, like type arguments and
574      * array specifiers.
575      */
576     @Nonnull
577     private AbstractJClass _parseSuffix (@Nonnull final AbstractJClass clazz)
578     {
579       if (_idx == _s.length ())
580         return clazz; // hit EOL
581 
582       final char ch = _s.charAt (_idx);
583 
584       if (ch == '<')
585         return _parseSuffix (_parseArguments (clazz));
586 
587       if (ch == '[')
588       {
589         if (_s.charAt (_idx + 1) == ']')
590         {
591           _idx += 2;
592           return _parseSuffix (clazz.array ());
593         }
594         throw new IllegalArgumentException ("Expected ']' but found " + _s.substring (_idx + 1));
595       }
596 
597       return clazz;
598     }
599 
600     /**
601      * Skips whitespaces
602      */
603     private void _skipWs ()
604     {
605       while (Character.isWhitespace (_s.charAt (_idx)) && _idx < _s.length ())
606         _idx++;
607     }
608 
609     /**
610      * Parses '&lt;T1,T2,...,Tn>'
611      * 
612      * @return the index of the character next to '>'
613      */
614     @Nonnull
615     private AbstractJClass _parseArguments (@Nonnull final AbstractJClass rawType)
616     {
617       if (_s.charAt (_idx) != '<')
618         throw new IllegalArgumentException ();
619       _idx++;
620 
621       final List <AbstractJClass> args = new ArrayList <AbstractJClass> ();
622 
623       while (true)
624       {
625         args.add (parseTypeName ());
626         if (_idx == _s.length ())
627           throw new IllegalArgumentException ("Missing '>' in " + _s);
628         final char ch = _s.charAt (_idx);
629         if (ch == '>')
630           return rawType.narrow (args.toArray (new AbstractJClass [args.size ()]));
631 
632         if (ch != ',')
633           throw new IllegalArgumentException (_s);
634         _idx++;
635       }
636     }
637   }

639   /**
640    * References to existing classes.
641    * <p>
642    * JReferencedClass is kept in a pool so that they are shared. There is one
643    * pool for each JCodeModel object.
644    * <p>
645    * It is impossible to cache JReferencedClass globally only because there is
646    * the _package() method, which obtains the owner JPackage object, which is
647    * scoped to JCodeModel.
648    */
649   private class JReferencedClass extends AbstractJClass implements IJDeclaration
650   {
651     private final Class <?> _class;
652 
653     JReferencedClass (@Nonnull final Class <?> _clazz)
654     {
655       super (JCodeModel.this);
656       _class = _clazz;
657       assert !_class.isArray ();
658     }
659 
660     @Override
661     public String name ()
662     {
663       return _class.getSimpleName ();
664     }
665 
666     @Override

668     {
669       return NameUtilities.getFullName (_class);
670     }
671 
672     @Override
673     public String binaryName ()
674     {

676     }
677 
678     @Override
679     public AbstractJClass outer ()
680     {
681       final Class <?> p = _class.getDeclaringClass ();
682       if (p == null)
683         return null;
684       return ref (p);
685     }
686 
687     @Override
688     @Nonnull
689     public JPackage _package ()
690     {
691       final String name = fullName ();
692 
693       // this type is array
694       if (name.indexOf ('[') != -1)
695         return JCodeModel.this._package ("");
696 
697       // other normal case
698       final int idx = name.lastIndexOf ('.');
699       if (idx < 0)
700         return JCodeModel.this._package ("");
701       return JCodeModel.this._package (name.substring (0, idx));
702     }

704     @Override
705     public AbstractJClass _extends ()
706     {
707       final Class <?> sp = _class.getSuperclass ();
708       if (sp == null)
709       {
710         if (isInterface ())
711           return owner ().ref (Object.class);
712         return null;
713       }
714       return ref (sp);
715     }
